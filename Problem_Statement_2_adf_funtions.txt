def currentTimeStamp:
day = utc_now(“dd”)
month = utc_now(“MM”)
year = utc_now(“yyyy”)
hh=utc_now(“hh”)
mm=utc_now(“mm”)
ss=utc_now(“ss”)
currentTimeStamp = year + month + day + hh + mm + ss
return currentTimeStamp

def DateFromDaysSince2:
addDays = addDays(toDate('2022-05-01'), -11) -> toDate('2016-08-09')
addDaysDate= toDate(addDays)
return addDaysDate

def MinutesFromTime:
